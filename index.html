# --- CONFIGURATION --- $CheminCentral = "C:\Users\Police\Documents\Import_SnipeIT.csv" # --- RÉCUPÉRATION DES DONNÉES --- $Hostname = $env:COMPUTERNAME $ComputerSystem = Get-CimInstance Win32_ComputerSystem $Bios = Get-CimInstance Win32_BIOS $Cpu = (Get-CimInstance Win32_Processor).Name.Trim() $RamGB = [Math]::Round($ComputerSystem.TotalPhysicalMemory / 1GB, 2) $OS = (Get-CimInstance Win32_OperatingSystem).Caption # Génération d'un Asset Tag (ex: AST-12345) $RandomAssetTag = "AST-" + (Get-Random -Minimum 10000 -Maximum 99999) # Note formatée $Note = "Machine : $Hostname | CPU : $Cpu | RAM : $RamGB Go | OS : $OS" # --- CRÉATION DE L'OBJET (Format Snipe-IT) --- # Note : Les noms des colonnes à gauche doivent correspondre aux en-têtes Snipe-IT $ObjetInventaire = [PSCustomObject]@{ "Asset Tag" = $RandomAssetTag # Requis "Serial Number" = $Bios.SerialNumber # "Serial" devient "Serial Number" pour Snipe "Model Name" = $ComputerSystem.Model # Snipe préfère "Model Name" pour créer le modèle s'il n'existe pas "Status" = "Ready to Deploy" # Utiliser un statut qui existe dans votre Snipe-IT (ex: "Actif" ou "Ready to Deploy") "Location" = "Belfort" # Correspond à "Default Location" "Notes Biens" = $Note # "Note" devient "Notes" (avec un S) } # --- EXPORTATION --- try { # On utilise l'encodage UTF8 et le délimiteur virgule (souvent préféré par l'importateur Snipe-IT) $ObjetInventaire | Export-Csv -Path $CheminCentral -NoTypeInformation -Delimiter "," -Encoding utf8 -Append Write-Host "Ligne ajoutée avec succès pour Snipe-IT !" -ForegroundColor Green } catch { Write-Host "Erreur : $($_.Exception.Message)" -ForegroundColor Red }
